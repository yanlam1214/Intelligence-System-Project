130    def filter_suit(self, suit: Suit) -> Iterable[Card]:

132        results: list[Card] = list(filter(lambda x: x.suit is suit, self.get_cards()))
133        return results

140    @abstractmethod

142        pass
143

150    def __contains__(self, item: Any) -> bool:
